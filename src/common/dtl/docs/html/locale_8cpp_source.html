00367                            g_free (unichar2BeginRegionCode);
00368          #else
00369                            gchar* charBeginRegionCode = g_utf16_to_utf8 ((gunichar2*) beginRegionCode, -1, NULL, &wsize, NULL);
00370          #endif
00371                            region.assign (charBeginRegionCode, (size_t)wsize);
00372                            g_free (charBeginRegionCode);
