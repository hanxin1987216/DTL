}

                                                {
    bool rightDay = false;
    switch (_mode) {
        case EVERYDAY: {
            rightDay = true;
            break;
        }
        case WEEK: {
            for(std::vector<int>::const_iterator it = _date.begin()
                ;it != _date.end()
                ;++it) {
                if (checkWeek(date.getDayOfWeek(), *it)) {
                    rightDay = true;
                    break; //break for(..)
                }
            }
            break;
        }
        case MONTH: {
            for(std::vector<int>::const_iterator it = _date.begin()
                ;it != _date.end()
                ;++it) {
                if (static_cast<int>(date.getDay()) == *it) {
                    rightDay = true;
                    break; //break for(..)
                }
            }
            break;
        }
        default: {
            assert(false);
        }
    }
    //比较当前时间
    if (!rightDay) {
        return false;
    } else if (cmpDateWithTimeStr(date, _startTime) <= 0
             && cmpDateWithTimeStr(date, _endTime) > 0) {
        return true;
    }
    return false;
}
