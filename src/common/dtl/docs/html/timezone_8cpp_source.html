00101 loadTimeZoneInfo (unsigned int year, 
00102                                       int64& UTCOffset, 
00103                                       int64& UTCOffsetWithDls,
00104                                       String& std_name)
00105 {
00106 #ifdef __WINDOWS__
00107          TIME_ZONE_INFORMATION tz_info;
00108          int tz_id;
00109 
00110          tz_id = ::GetTimeZoneInformation (&tz_info);
00111          if (tz_id == TIME_ZONE_ID_INVALID)
00112                   return false;
00113 
00114          if ((year <= 1601) || (year > 30827)) {
00115                   // According to MSDN, the MS time functions can't handle dates outside
00116                   // this interval.
00117                   return true;
00118          }
00119 
00120          std_name = tz_info.StandardName;
00121 
00122          UTCOffset = ((tz_info.Bias + tz_info.StandardBias) * -60000000LL);
00123          UTCOffsetWithDls = UTCOffset + ((tz_info.DaylightBias - tz_info.StandardBias) * -60000000LL);
00124 
00125          return true;
00126 #else
00127 
00128          struct tm start, tt;
00129          time_t t;
00130 
00131          long int gmtoff;
00132          int is_daylight = 0, day;
00133          char tzone [64];
00134 
00135          memset (&start, 0, sizeof (start));
00136 
00137          start.tm_mday = 1;
00138          start.tm_year = year-1900;
00139 
00140          t = mktime (&start);
00141 
00142          if ((year < 1970) || (year > 2037) || (t == -1)) {
00143                   t = time (NULL);
00144                   tt = *localtime (&t);
00145                   strftime (tzone, sizeof (tzone), "%Z", &tt);
00146                   std_name = tzone;
00147                   UTCOffset = gmt_offset (&start, t);
00148                   UTCOffsetWithDls = UTCOffset + (tt.tm_isdst * -1000000LL);
00149                   return true;
00150          }
00151 
00152          gmtoff = gmt_offset (&start, t);
00153 
00154          /* For each day of the year, calculate the tm_gmtoff. */
00155          for (day = 0; day < 365; day++) {
00156 
00157                   t += 3600*24;
00158                   tt = *localtime (&t);
00159 
00160                   /* Daylight saving starts or ends here. */
00161                   if (gmt_offset (&tt, t) != gmtoff) {
00162                            struct tm tt1;
00163                            time_t t1;
00164 
00165                            /* Try to find the exact hour when daylight saving starts/ends. */
00166                            t1 = t;
00167                            do {
00168                                     t1 -= 3600;
00169                                     tt1 = *localtime (&t1);
00170                            } while (gmt_offset (&tt1, t1) != gmtoff);
00171 
00172                            /* Try to find the exact minute when daylight saving starts/ends. */
00173                            do {
00174                                     t1 += 60;
00175                                     tt1 = *localtime (&t1);
00176                            } while (gmt_offset (&tt1, t1) == gmtoff);
00177                            t1+=gmtoff;
00178                            strftime (tzone, sizeof (tzone), "%Z", &tt);
00179                            
00180                            /* Write data, if we're already in daylight saving, we're done. */
00181                            if (!is_daylight) {
00182                                     std_name = tzone;
00183                                     UTCOffset = 0;
00184                                     UTCOffsetWithDls = (gint64)t1 * 1000000L;
00185                                     return true;
00186                            } else {
00187                                     UTCOffset = (gint64)t1 * 1000000L;
00188                                     UTCOffsetWithDls = 0;
00189                                     is_daylight = 1;
00190                            }
00191 
00192                            /* This is only set once when we enter daylight saving. */
00193                            UTCOffset = (gint64)gmtoff * 1000000L;
00194                            UTCOffsetWithDls = (gint64)(gmt_offset (&tt, t) - gmtoff) * 1000000L;
00195                            gmtoff = gmt_offset (&tt, t);
00196                   }
00197          }
00198 
00199          if (!is_daylight) {
00200                   strftime (tzone, sizeof (tzone), "%Z", &tt);
00201                   std_name = tzone;
00202                   UTCOffset = (gint64)gmtoff * 1000000L;
00203                   UTCOffsetWithDls = 0;
00204          }
00205 
00206          return true;
